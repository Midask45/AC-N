import java.util.*;

public class P1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // a) Roots of Quadratic Equation
        System.out.println("Enter coefficients a, b, c of quadratic equation:");
        double a = sc.nextDouble();
        double b = sc.nextDouble();
        double c = sc.nextDouble();

        double d = b * b - 4 * a * c;

        if (d >= 0) {
            double r1 = (-b + Math.sqrt(d)) / (2 * a);
            double r2 = (-b - Math.sqrt(d)) / (2 * a);
            System.out.println("Roots are: " + r1 + " , " + r2);
        } else {
            System.out.println("Roots are imaginary");
        }

        // b) Multiply two arrays
        System.out.println("\nEnter size of arrays:");
        int n = sc.nextInt();

        int[] x = new int[n];
        int[] y = new int[n];
        int[] z = new int[n];

        System.out.println("Enter elements of first array:");
        for (int i = 0; i < n; i++)
            x[i] = sc.nextInt();

        System.out.println("Enter elements of second array:");
        for (int i = 0; i < n; i++)
            y[i] = sc.nextInt();

        for (int i = 0; i < n; i++)
            z[i] = x[i] * y[i];

        System.out.println("Product of arrays: " + Arrays.toString(z));

        // c) Bubble Sort Ascending & Descending
        System.out.println("\nEnter number of elements to sort:");
        int m = sc.nextInt();

        int[] arr = new int[m];
        System.out.println("Enter elements:");
        for (int i = 0; i < m; i++)
            arr[i] = sc.nextInt();

        // Ascending order
        int[] asc = arr.clone();
        for (int i = 0; i < asc.length - 1; i++) {
            for (int j = 0; j < asc.length - 1 - i; j++) {
                if (asc[j] > asc[j + 1]) {
                    int temp = asc[j];
                    asc[j] = asc[j + 1];
                    asc[j + 1] = temp;
                }
            }
        }
        System.out.println("Sorted Ascending: " + Arrays.toString(asc));

        // Descending order
        int[] desc = arr.clone();
        for (int i = 0; i < desc.length - 1; i++) {
            for (int j = 0; j < desc.length - 1 - i; j++) {
                if (desc[j] < desc[j + 1]) {
                    int temp = desc[j];
                    desc[j] = desc[j + 1];
                    desc[j + 1] = temp;
                }
            }
        }
        System.out.println("Sorted Descending: " + Arrays.toString(desc));

        sc.close();
    }
}

import java.util.Scanner;

class Complex {
    private double r, i;

    // Constructor
    Complex(double r, double i) {
        this.r = r;
        this.i = i;
    }

    // Addition
    Complex add(Complex c) {
        return new Complex(this.r + c.r, this.i + c.i);
    }

    // Subtraction
    Complex sub(Complex c) {
        return new Complex(this.r - c.r, this.i - c.i);
    }

    // Comparison
    boolean comp(Complex c) {
        return (this.r == c.r && this.i == c.i);
    }

    // Display
    void display() {
        System.out.println(r + " + " + i + "i");
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input first complex number
        System.out.print("Enter real and imaginary part of first complex number: ");
        double r1 = sc.nextDouble();
        double i1 = sc.nextDouble();

        // Input second complex number
        System.out.print("Enter real and imaginary part of second complex number: ");
        double r2 = sc.nextDouble();
        double i2 = sc.nextDouble();

        Complex n1 = new Complex(r1, i1);
        Complex n2 = new Complex(r2, i2);

        Complex sum = n1.add(n2);
        Complex diff = n1.sub(n2);

        System.out.print("Sum = ");
        sum.display();

        System.out.print("Difference = ");
        diff.display();

        System.out.println("Both complex numbers are equal: " + n1.comp(n2));

        sc.close();
    }
}

import java.util.Scanner;

class SC {

    // Compare full strings
    static boolean usrstrcmp(String s1, String s2) {
        if (s1.length() != s2.length())
            return false;

        for (int i = 0; i < s1.length(); i++)
            if (s1.charAt(i) != s2.charAt(i))
                return false;

        return true;
    }

    // Compare first n characters
    static boolean usrstrcmp(String s1, String s2, int n) {
        int len = Math.min(n, Math.min(s1.length(), s2.length()));

        for (int i = 0; i < len; i++)
            if (s1.charAt(i) != s2.charAt(i))
                return false;

        return true;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter first string: ");
        String s1 = sc.nextLine();

        System.out.print("Enter second string: ");
        String s2 = sc.nextLine();

        System.out.print("Enter number of characters to compare: ");
        int n = sc.nextInt();

        System.out.println("\nFull string comparison result: "
                + usrstrcmp(s1, s2));

        System.out.println("First " + n + " characters comparison result: "
                + usrstrcmp(s1, s2, n));

        sc.close();
    }
}

import java.util.Scanner;

// Base abstract class
abstract class Bank {
    String name;
    int accNo;
    double balance;

    // Read customer details
    void read(Scanner sc) {
        System.out.print("Enter Name: ");
        name = sc.nextLine();

        System.out.print("Enter Account Number: ");
        accNo = sc.nextInt();

        System.out.print("Enter Balance: ");
        balance = sc.nextDouble();
        sc.nextLine(); // clear buffer
    }

    // Display details
    void display() {
        System.out.printf(
            "Name: %s | Acc No: %d | Balance: %.2f | Interest: %.2f\n",
            name, accNo, balance, calculateInterest()
        );
    }

    // Abstract method
    abstract double calculateInterest();
}

// Derived class: City Bank
class CityBank extends Bank {
    double calculateInterest() {
        return balance * 0.06; // 6%
    }
}

// Derived class: SBI Bank
class SBIBank extends Bank {
    double calculateInterest() {
        return balance * 0.07; // 7%
    }
}

// Derived class: Canara Bank
class CanaraBank extends Bank {
    double calculateInterest() {
        return balance * 0.065; // 6.5%
    }
}

// Main class
public class P6 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        Bank[] banks = new Bank[3];
        banks[0] = new CityBank();
        banks[1] = new SBIBank();
        banks[2] = new CanaraBank();

        String[] bankNames = {"City Bank", "SBI Bank", "Canara Bank"};

        for (int i = 0; i < banks.length; i++) {
            System.out.println("\nEnter details for " + bankNames[i] + " customer:");
            banks[i].read(sc);
        }

        System.out.println("\n--- Interest Details of All Account Holders ---");
        for (Bank b : banks) {
            b.display();
        }

        sc.close();
    }
}

// Write a multi-threaded java program to illustrate producer consumer problem.

class P7{
    int[] buf=new int[5];
    int in=0, out=0, count=0;
    synchronized void put(int n) throws InterruptedException{
        while(buf.length==count) wait();
        buf[in]=n;
        in=(in+1)%buf.length;
        count++;
        System.out.println("\nProduced: "+n);
        notifyAll();
    }
    synchronized int get() throws InterruptedException{
        while(count==0) wait();
        int item=buf[out];
        out=(out+1)%buf.length;
        count--;
        System.out.println("\nConsumed: "+item);
        notifyAll();
        return item;
    }
    public static void main(String[] args){
        P7 pc=new P7();
        new Thread(()->{
            try{
                for(int i=1;i<=10;i++){
                    pc.put(i);
                    Thread.sleep(500);
                }
            } catch (InterruptedException ignored) {}
        }).start();
        new Thread(()->{
            try{
                while(true){
                    pc.get();
                    Thread.sleep(1000);
                }
            } catch (InterruptedException ignored) {}
        }).start();
    }
}

import java.util.Scanner;

interface Compute {
    double convert(double v);
}

class GB2B implements Compute {
    public double convert(double gb) {
        return gb * 1073741824; // 1 GB = 2^30 bytes
    }
}

class E2R implements Compute {
    public double convert(double euro) {
        return euro * 90.85; // Euro to Rupees
    }
}

public class P9 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        Compute gbConverter = new GB2B();
        Compute euroConverter = new E2R();

        System.out.print("Enter value in Giga Bytes: ");
        double gb = sc.nextDouble();

        System.out.print("Enter value in Euros: ");
        double euro = sc.nextDouble();

        System.out.println("\nConversions:");
        System.out.println(gb + " GB = " + gbConverter.convert(gb) + " Bytes");
        System.out.println(euro + " Euros = " + euroConverter.convert(euro) + " Rupees");

        sc.close();
    }
}

import java.sql.*;

public class JDBCCRUD {

    static final String URL = "jdbc:mysql://localhost:3306/jdbc_demo";
    static final String USER = "root";
    static final String PASS = "";

    public static void main(String[] args) {

        try {
            Class.forName("com.mysql.cj.jdbc.Driver");

            Connection con = DriverManager.getConnection(URL, USER, PASS);
            System.out.println("Connected to database");

            // CREATE
            String insert = "INSERT INTO student(name, age) VALUES (?, ?)";
            PreparedStatement ps = con.prepareStatement(insert);
            ps.setString(1, "Alice");
            ps.setInt(2, 21);
            ps.executeUpdate();
            System.out.println("Record inserted");

            // READ
            Statement st = con.createStatement();
            ResultSet rs = st.executeQuery("SELECT * FROM student");

            System.out.println("Student Records:");
            while (rs.next()) {
                System.out.println(
                        rs.getInt("id") + " " +
                        rs.getString("name") + " " +
                        rs.getInt("age")
                );
            }

            // UPDATE
            ps = con.prepareStatement(
                    "UPDATE student SET age=? WHERE name=?");
            ps.setInt(1, 22);
            ps.setString(2, "Alice");
            ps.executeUpdate();
            System.out.println("Record updated");

            // DELETE
            ps = con.prepareStatement(
                    "DELETE FROM student WHERE name=?");
            ps.setString(1, "Alice");
            ps.executeUpdate();
            System.out.println("Record deleted");

            con.close();

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
